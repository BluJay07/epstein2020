#Bargaining with RUS - Intro
country_event = {
	id = ins.1
	title = ins.1.t
	desc = ins.1.d
	picture = RUS_Menu
	trigger = {
	}

	is_triggered_only = yes
	
	option = {
		name = ins.1.a
		if = {
			limit = {
				OR = {
				has_government = autocratic
				has_government = republican
				}
			}
				country_event = { id = ins.2 }
				add_to_variable = {
				var = ROOT.Relation_RUS
				value = 1
				}

			
			else = {
				

				country_event = { id = ins.3 }

		}
	}
}
	option = {
		name = ins.1.b
		if = {
			limit = {
				OR = {
				has_government = autocratic

				}

				var = ROOT.Relation_RUS
				value = 1
				}

			
			else = {
				

				country_event = { id = ins.3 }

		}
	}
	}
}
