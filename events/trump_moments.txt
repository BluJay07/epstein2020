# Pence ultimately wins the war with Trump
country_event = {
	id = maga.2
	title = maga.2.t
	desc = maga.2.d
	picture = event_NPA_victory
	trigger = {
		tag = NPA
		owns_state = 215
		has_war = no
	}
	mean_time_to_happen = {
		days = 1
	}
	fire_only_once = yes
	option = {
		name = maga.2.a
		set_cosmetic_tag = NPA_nation
		every_owned_state = { 
			limit = { is_core_of = USA }
			add_core_of = NPA
		}
		add_popularity = {
			ideology = republican
			popularity = 0.2
		}
		hidden_effect = {
		news_event = { id = maga.3 days = 1 } 
		every_owned_state = { 
			limit = { is_core_of = USA }
			remove_core_of = USA
		}
		if = {
			limit = {
				has_idea = NPA_Motivation_High
			}
			remove_ideas = NPA_Motivation_High
		}
		}
	}
}
